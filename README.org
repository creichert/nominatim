#+BEGIN_SRC haskell
{-# LANGUAGE OverloadedStrings #-}

import Web.Nominatim

main :: IO ()
  let hdrs = [("User-Agent", BL.pack "Nominatim Client")]
  nominatim hdrs [ Format Json
                 , Query "1234 Awesome Dr, Houston, TX"
                 , AddressDetails True
                 , Limit 10
                 ]
